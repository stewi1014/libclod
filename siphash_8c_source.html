21sip_round(clod_sip64_state *state) {
22 state->_v0 += state->_v1;
23 state->_v1 = state->_v1 << 13 | state->_v1 >> 51;
24 state->_v1 ^= state->_v0;
25 state->_v0 = state->_v0 << 32 | state->_v0 >> 32;
26 state->_v2 += state->_v3;
27 state->_v3 = state->_v3 << 16 | state->_v3 >> 48;
28 state->_v3 ^= state->_v2;
29 state->_v0 += state->_v3;
30 state->_v3 = state->_v3 << 21 | state->_v3 >> 43;
31 state->_v3 ^= state->_v0;
32 state->_v2 += state->_v1;
33 state->_v1 = state->_v1 << 17 | state->_v1 >> 47;
34 state->_v1 ^= state->_v2;
35 state->_v2 = state->_v2 << 32 | state->_v2 >> 32;

38read_uint64(const unsigned char *restrict data, const size_t data_size) {
39 uint64_t r = 0;
40 switch (data_size) {
41 default: r |= (uint64_t)data[7] << 7 * 8; __attribute__((fallthrough));
42 case 7: r |= (uint64_t)data[6] << 6 * 8; __attribute__((fallthrough));
43 case 6: r |= (uint64_t)data[5] << 5 * 8; __attribute__((fallthrough));
44 case 5: r |= (uint64_t)data[4] << 4 * 8; __attribute__((fallthrough));
45 case 4: r |= (uint64_t)data[3] << 3 * 8; __attribute__((fallthrough));
46 case 3: r |= (uint64_t)data[2] << 2 * 8; __attribute__((fallthrough));
47 case 2: r |= (uint64_t)data[1] << 1 * 8; __attribute__((fallthrough));
48 case 1: r |= (uint64_t)data[0] << 0 * 8; __attribute__((fallthrough));
49 case 0: return r;

52clod_sip64_state clod_sip64_add(clod_sip64_state state, const void *data, const size_t size) {
53 if (size == 0) return state;
54 const unsigned char* restrict in = data;
55 set_data_size(state, data_size(state) + size);
56
57 if (remaining(state) + size < 8) {
58 memcpy(state._buf + remaining(state), in, size);
59 set_remaining(state, remaining(state) + size);
60 return state;
61 }
62
63 uint64_t d, off = 0;
64 if (remaining(state) > 0) {
65 d = read_uint64(state._buf, remaining(state));
66 d |= read_uint64(in, 8 - remaining(state)) << remaining(state) * 8;
67 off = 8 - remaining(state);
68 set_remaining(state, 0);
69 } else {
70 d = read_uint64(in, 8);
71 off = 8;
72 }
73
74 state._v3 ^= d;
75 sip_round(&state);
76 sip_round(&state);
77 state._v0 ^= d;
78
79 while (off + 8 <= size) {
80 d = read_uint64(in + off, 8);
81 off += 8;
82
83 state._v3 ^= d;
84 sip_round(&state);
85 sip_round(&state);
86 state._v0 ^= d;
87 }
88
89 if (off < size) {
90 memcpy(state._buf, in + off, size - off);
91 set_remaining(state, size - off);
92 }
93
94 return state;
95}
96uint64_t clod_sip64_finalise(clod_sip64_state state) {
97 uint64_t d = read_uint64(state._buf, remaining(state));
98 d |= (uint64_t)(data_size(state)) << 56;
99 state._v3 ^= d;
100 sip_round(&state);
101 sip_round(&state);
102 state._v0 ^= d;
103 state._v2 ^= 0xee;
104 sip_round(&state);
105 sip_round(&state);
106 sip_round(&state);
107 sip_round(&state);
108 return state._v0 ^ state._v1 ^ state._v2 ^ state._v3;
