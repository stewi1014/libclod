87probe(const struct clod_table *t, const struct table_position pos, const void *key, const size_t key_size) {
88 assert(pos.index < t->table_size);
89
90 size_t available = INDEX_NIL;
91 for (size_t i = 0; i < t->table_size; i++) {
92 const size_t index = (pos.index + i) % t->table_size;
93
94 if (t->control[index] == CTL_EMPTY) {
95 return (struct probe){
96 .existing = INDEX_NIL,
97 .available = available != INDEX_NIL ? available : index
98 };
99 }
100
101 if (t->control[index] == pos.ctl && key_equal(t, index, key, key_size)) {
102 return (struct probe){
103 .existing = index,
104 .available = INDEX_NIL
105 };
106 }
107
108 if (t->control[index] == CTL_REMOVED && available == INDEX_NIL) {
109 available = index;
110 }
111 }
112
113 return (struct probe){
114 .existing = INDEX_NIL,
115 .available = available
116 };
117}
118static const void *insert(struct clod_table *t, const bool replace, const void *element, const size_t key_size) {
119 assert(t->elem_count < t->table_size);
120
121 auto const pos = get_position(t, element, key_size);
122 auto const res = probe(t, pos, element, key_size);
123
124 if (res.existing != INDEX_NIL) {
125 if (replace) {
126 const void *previous = t->elements[res.existing].element;
127 t->elements[res.existing].element = element;
128 t->elements[res.existing].key_size = key_size;
129 return previous;
130 } else {

132 }
133 }
134
135 assert(res.available != INDEX_NIL);
136 t->elem_count++;

138 t->control[res.available] = pos.ctl;
139 t->elements[res.available].element = element;
140 t->elements[res.available].key_size = key_size;
141 return nullptr;
142}
143static bool rebuild(struct clod_table *t, const size_t new_table_size) {

234bool clod_table_iter(const struct clod_table *t, struct clod_table_iter *iter) {
235 while (iter->_internal < t->table_size) {
236 const size_t index = (iter->_internal + t->cursor) % t->table_size;
237 if (CTL_IS_OCCUPIED(t->control[index])) {
238 auto const res = t->elements[index];
239 iter->element = (void*)res.element;
240 iter->key_size = res.key_size;
241 iter->_internal++;
242 return true;
243 }
244 iter->_internal++;
245 }
246 iter->_internal = 0;
247 iter->element = nullptr;
248 iter->key_size = 0;
